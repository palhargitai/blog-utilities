9   /**
10   * Describes a utility that can read settings. It loads a 'default' file from
11   * the classpath and provides some utility methods to access the settings.
12   * 
13   * 
14   * @author Pal Hargitai
15   */
16  public enum SettingsUtil {
17  
18  	/** Default instance. */
19  	INSTANCE;
20  
21  	/** A logger. */
22  	private static final Logger LOGGER = Logger.getLogger(SettingsUtil.class.getCanonicalName());
23  	/** Properties file location, contains actual settings. */
24  	private static String PROPERTIES_LOCATION = "/settings.properties";
25  
26  	/** Properties containing the settings. */
27  	private volatile Properties settings = null;
28  
29  	/**
30  	 * Updates the properties location.
31  	 * 
32  	 * @param propertiesLocation
33  	 *            The new properties location.
34  	 */
35  	protected void updatePropertiesLocation(String propertiesLocation) {
36  		SettingsUtil.PROPERTIES_LOCATION = propertiesLocation;
37  	}
38  
39  	/**
40  	 * Gets a value from the settings, if the value is not set, return the
41  	 * defaultValue.
42  	 * 
43  	 * @param key
44  	 *            The key to return the settings for.
45  	 * @param defaultValue
46  	 *            The default value.
47  	 * @return The value associated with the key from the properties, or, if not
48  	 *         set, the defaultValue.
49  	 */
50  	private String getValueDelegate(String key, String defaultValue) {
51  		SettingsUtil.LOGGER.fine(String.format("Getting property '%s' with default value '%s'.", key, defaultValue));
52  		final Properties properties = this.getSettings();
53  		final String result;
54  		if (properties.containsKey(key)) {
55  			result = properties.getProperty(key);
56  		} else {
57  			result = defaultValue;
58  		}
59  		SettingsUtil.LOGGER.fine(String.format("Resolved property value '%s' for property '%s'.", result, key));
60  		return result;
61  	}
62  
63  	/**
64  	 * Gets the settings, if the settings have not been retrieved before,
65  	 * retrieves them.
66  	 * 
67  	 * @return The settings.
68  	 */
69  	private Properties getSettings() {
70  		Properties result = this.settings;
71  		if (result == null) {
72  			synchronized (this) {
73  				result = this.settings;
74  				if (result == null) {
75  					this.settings = result = new Properties();
76  					SettingsUtil.initializeProperties(this.settings, SettingsUtil.PROPERTIES_LOCATION);
77  				}
78  			}
79  		}
80  		return result;
81  	}
82  
83  	/** Resets the properties. Only for testing! */
84  	protected void resetProperties() {
85  		synchronized (this) {
86  			this.settings = null;
87  		}
88  	}

90  	/**
91  	 * Initializes the properties with the values from the given location.
92  	 * 
93  	 * @param properties
94  	 *            The properties.
95  	 */
96  	private static void initializeProperties(Properties properties, String location) {
97  		InputStream resourceStream = null;
98  		// First attempt to load the resource from the context classloader.
99  		final ClassLoader contextClassLoader = Thread.currentThread().getContextClassLoader();
100 		if (contextClassLoader != null) {
101 			resourceStream = contextClassLoader.getResourceAsStream(location);
102 		}
103 		// If the first attempt failed, use the settings class to resolve the
104 		// resource.
105 		if (resourceStream == null) {
106 			resourceStream = SettingsUtil.class.getResourceAsStream(location);
107 		}
108 		// Load the resource
109 		try {
110 			if (resourceStream != null) {
111 				properties.load(resourceStream);
112 			}
113 		} catch (IOException e) {
114 			// We just log the exception.
115 			SettingsUtil.LOGGER.warning(String.format("Could not read from '%s' to properties.", location));
116 			SettingsUtil.LOGGER.log(Level.FINE, String.format("Exception occured while reading '%s'.", location), e);
117 		} finally {
118 			if (resourceStream != null) {
119 				try {
120 					resourceStream.close();
121 				} catch (IOException e) {
122 					SettingsUtil.LOGGER.warning(String.format("An exception occured while closing the stream for '%s'.", location));
123 					SettingsUtil.LOGGER.log(Level.FINE, String.format("Exception occured closing stream for '%s'.", location), e);
124 					// Don't rethrow the exception, as it may overwrite the
125 					// originating exception!
126 				}
127 			}
128 		}
129 	}

131 	/**
132 	 * Gets a value from the settings, if the value is not set, return the
133 	 * defaultValue.
134 	 * 
135 	 * @param key
136 	 *            The key to return the settings for.

140 	 *         set, the defaultValue.
141 	 */
142 	public static String getValue(String key, String defaultValue) {
143 		// Delegate to instance method.
144 		return SettingsUtil.INSTANCE.getValueDelegate(key, defaultValue);
145 	}
